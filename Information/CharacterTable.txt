CREATE TABLE CharacterDetails (
    Id SERIAL PRIMARY KEY,
    userKey VARCHAR(255) NOT NULL,
    characterUid UUID NOT NULL,
    Name VARCHAR(255) NOT NULL,
    Server VARCHAR(255),
    serverIdentifier VARCHAR(255),
    Cosmetics JSONB,
    UniversalColors JSONB,
    isMale BOOLEAN,
    seenIntro BOOLEAN,
    skippedTutorial BOOLEAN
);

CREATE TABLE CharacterDetails
(
    id SERIAL PRIMARY KEY,
    userKey VARCHAR(255) NOT NULL,
    characterUid VARCHAR(255) NOT NULL,
    name VARCHAR(255),
    server VARCHAR(255),
    serverIdentifier VARCHAR(255),
    cosmetics JSONB,
    universalColors JSONB,
    isMale BOOLEAN,
    seenIntro BOOLEAN,
    skippedTutorial BOOLEAN,
    headId VARCHAR(255),
    headPrefab VARCHAR(255),
    headColorPrimaryR FLOAT,
    headColorPrimaryG FLOAT,
    headColorPrimaryB FLOAT,
    headColorPrimaryA FLOAT,
    headColorSecondaryR FLOAT,
    headColorSecondaryG FLOAT,
    headColorSecondaryB FLOAT,
    headColorSecondaryA FLOAT,
    headColorTertiaryR FLOAT,
    headColorTertiaryG FLOAT,
    headColorTertiaryB FLOAT,
    headColorTertiaryA FLOAT,
    headColorSpecR FLOAT,
    headColorSpecG FLOAT,
    headColorSpecB FLOAT,
    headColorSpecA FLOAT,
    headHealth FLOAT,
    bodyId VARCHAR(255),
    bodyPrefab VARCHAR(255),
    bodyColorPrimaryR FLOAT,
    bodyColorPrimaryG FLOAT,
    bodyColorPrimaryB FLOAT,
    bodyColorPrimaryA FLOAT,
    bodyColorSecondaryR FLOAT,
    bodyColorSecondaryG FLOAT,
    bodyColorSecondaryB FLOAT,
    bodyColorSecondaryA FLOAT,
    bodyColorTertiaryR FLOAT,
    bodyColorTertiaryG FLOAT,
    bodyColorTertiaryB FLOAT,
    bodyColorTertiaryA FLOAT,
);
